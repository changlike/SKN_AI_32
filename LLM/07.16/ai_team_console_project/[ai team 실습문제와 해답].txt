문제 1 — Reviewer 노드 추가하기
요구사항
code/ai_team.py를 복사해 code/ex_review.py를 만드세요.

TeamState에 review, final 키를 추가한다.
writer 노드 뒤에 두 노드를 추가:
reviewer — report의 사실성·가독성을 평가한 코멘트를 review에 저장.
finalize — 그 코멘트를 반영한 최종본을 final에 작성.
엣지: writer → reviewer → finalize → END.
기대 결과: 실행 시 final 리포트가 출력되고, review 코멘트도 함께 확인된다.
왜 하나? Writer가 쓴 초안을 검토(reviewer) → 반영(finalize) 하는 단계를 더하면 품질이 올라갑니다. 
"초안 → 리뷰 → 최종본"은 사람 조직의 문서 작성 과정 그대로죠. 
파이프라인에 노드를 이어 붙이기만 하면 된다는 점이 StateGraph의 장점입니다.

# code/ex_review.py
# 실행: python code/ex_review.py
import sys, pathlib
sys.path.append(str(pathlib.Path(__file__).resolve().parent))
from common import get_chat

from typing import TypedDict
from langchain_core.messages import SystemMessage, HumanMessage
from langgraph.graph import StateGraph, START, END

# 기존 노드·로더 재사용
from ch27_ai_team import (
    TeamState as BaseState,
    researcher, analyst, writer, load_competitor_text,
)

llm = get_chat(temperature=0.2)


class TeamState(BaseState):     # 기존 상태에 칸 2개 추가
    review: str    # Reviewer 코멘트
    final: str     # 코멘트 반영 최종본


def reviewer(state: TeamState) -> dict:
    """report의 사실성·가독성을 평가해 개선 코멘트를 review에 저장."""
    out = llm.invoke([
        SystemMessage("너는 보고서 검토자다. 주어진 리포트의 사실성·가독성·구성을 평가하고, "
                      "개선할 점을 3가지 이내로 구체적으로 지적하라. 다시 쓰지는 마라."),
        HumanMessage(f"[검토할 리포트]\n{state['report']}"),
    ]).content
    print("  [reviewer] 검토 코멘트 작성 완료")
    return {"review": out}


def finalize(state: TeamState) -> dict:
    """검토 코멘트를 반영해 최종본을 final에 작성."""
    out = llm.invoke([
        SystemMessage("너는 보고서 작성자다. 아래 [리포트]를 [검토 의견]에 따라 개선한 "
                      "최종본을 마크다운으로 작성하라."),
        HumanMessage(f"[리포트]\n{state['report']}\n\n[검토 의견]\n{state['review']}"),
    ]).content
    print("  [finalize] 코멘트 반영 최종본 작성 완료")
    return {"final": out}


def build_team():
    g = StateGraph(TeamState)
    g.add_node("researcher", researcher)
    g.add_node("analyst", analyst)
    g.add_node("writer", writer)
    g.add_node("reviewer", reviewer)       # 추가
    g.add_node("finalize", finalize)       # 추가

    g.add_edge(START, "researcher")
    g.add_edge("researcher", "analyst")
    g.add_edge("analyst", "writer")
    g.add_edge("writer", "reviewer")       # writer → reviewer
    g.add_edge("reviewer", "finalize")     # reviewer → finalize
    g.add_edge("finalize", END)            # finalize → 끝
    return g.compile()


if __name__ == "__main__":
    team = build_team()
    initial = {"raw": load_competitor_text(), "analysis": "", "report": "",
               "review": "", "final": ""}
    final = initial
    for state in team.stream(initial, stream_mode="values"):
        final = state

    print("\n===== 검토 코멘트 =====")
    print(final["review"])
    print("\n===== 최종본 =====")
    print(final["final"])

관찰 포인트

writer → reviewer → finalize 엣지 3줄로 검토·반영 단계가 추가됐습니다. 기존 노드는 그대로죠.
review(코멘트)와 final(최종본)이 각각 다른 칸에 저장돼, 둘 다 확인할 수 있습니다.
이게 06번 조건부 분기와 합쳐지면 "품질 미달이면 finalize → reviewer로 되돌리기" 같은 루프도 가능합니다.



문제 2 — 단계별 산출물 파일 저장
요구사항
문제 1의 팀에서 각 노드가 자기 결과를 파일로 저장하게 만드세요.

output/research.md (정리된 사실), output/analysis.md (분석),
output/report.md (초안), output/final.md (최종본).
검증: 실행 후 output/ 폴더에 4개 파일이 모두 생기고 내용이 비어 있지 않은지 확인.
왜 하나? 7번 절(중간 산출물 저장)의 응용입니다. 단계별 결과를 파일로 남기면, 리포트가 이상할 때 어느 단계가 문제였는지 파일을 열어 바로 추적할 수 있습니다.

# code/ch27_ex_save.py (문제 1의 ch27_ex_review.py 에 저장 추가)
import sys, pathlib
sys.path.append(str(pathlib.Path(__file__).resolve().parent))
from common import get_chat, ROOT

from typing import TypedDict
from langchain_core.messages import SystemMessage, HumanMessage
from langgraph.graph import StateGraph, START, END
from ch27_ai_team import (
    TeamState as BaseState, researcher, analyst, writer, load_competitor_text,
)
from ch27_ex_review import reviewer, finalize, TeamState   # 문제1 재사용

OUTPUT = ROOT / "output"


def save(name: str, content: str) -> None:
    """output/ 아래에 산출물을 저장한다(폴더 자동 생성)."""
    OUTPUT.mkdir(parents=True, exist_ok=True)
    (OUTPUT / name).write_text(content, encoding="utf-8")
    print(f"  [저장] {name} ({len(content)}자)")


# 각 노드를 감싸 결과를 저장하는 래퍼
def researcher_s(state): 
    upd = researcher(state); save("27_research.md", upd["raw"]); return upd

def analyst_s(state):
    upd = analyst(state); save("27_analysis.md", upd["analysis"]); return upd

def writer_s(state):
    upd = writer(state); save("27_report.md", upd["report"]); return upd

def finalize_s(state):
    upd = finalize(state); save("27_final.md", upd["final"]); return upd


def build_team():
    g = StateGraph(TeamState)
    g.add_node("researcher", researcher_s)
    g.add_node("analyst", analyst_s)
    g.add_node("writer", writer_s)
    g.add_node("reviewer", reviewer)        # 코멘트는 별도 저장 안 함(원하면 추가)
    g.add_node("finalize", finalize_s)
    g.add_edge(START, "researcher")
    g.add_edge("researcher", "analyst")
    g.add_edge("analyst", "writer")
    g.add_edge("writer", "reviewer")
    g.add_edge("reviewer", "finalize")
    g.add_edge("finalize", END)
    return g.compile()


if __name__ == "__main__":
    team = build_team()
    initial = {"raw": load_competitor_text(), "analysis": "", "report": "",
               "review": "", "final": ""}
    for state in team.stream(initial, stream_mode="values"):
        pass
    print(f"\n완료 — {OUTPUT} 폴더를 확인하세요.")
